% !TeX root = ./nine-square.tex
\documentclass[UTF8]{ctexart}

% Adjust paper and printer parameters in the following file
\input{paper-printer}

% Name the grid
\def\gridname{nine-square grid (九宮格)}

% Declare stamp dimensions and separations
\def\stampheight{13mm}\def\stampvsep{0mm}
\def\stampwidth {13mm}\def\stamphsep{0mm}

% Declare a tikz pic for stamping onto the grid
\providecolor{stampcolora}{cmyk:named}{magenta!66!yellow}
\providecolor{stampcolorb}{cmyk:named}{cyan!66!yellow}
% todo: minor dashed lines need v and h due to pattern
\tikzset{
  stamp majorline/.style = {
      stampcolora,
      line cap=rect,
      line width=0.3mm
    },
  stamp minorlineh/.style = {
      stampcolorb,
      line cap=butt,
      line width=0.15mm,
      dash pattern=on \stampwidth/3/3/2 off \stampwidth/3/3/2,
      dash phase=0.5*\stampwidth/3/3/2 % alternative: 1.5
    },
  stamp minorlinev/.style = {
      stampcolorb,
      line cap=butt,
      line width=0.15mm,
      dash pattern=on \stampheight/3/3/2 off \stampheight/3/3/2,
      dash phase=0.5*\stampheight/3/3/2 % alternative: 1.5
    },
  pics/stamp/.style={
      code={
          % Horizontal
          \draw [stamp minorlineh] (0,  \stampheight/3) -- (\stampwidth,  \stampheight/3);
          \draw [stamp minorlineh] (0,2*\stampheight/3) -- (\stampwidth,2*\stampheight/3);
          % Vertical
          \draw [stamp minorlinev] (  \stampwidth/3,0) -- (  \stampwidth/3,\stampheight);
          \draw [stamp minorlinev] (2*\stampwidth/3,0) -- (2*\stampwidth/3,\stampheight);
          % Horizontal
          \draw [stamp majorline ] (0,             0  ) -- (\stampwidth,             0  );
          \draw [stamp majorline ] (0,  \stampheight  ) -- (\stampwidth,  \stampheight  );
          % Vertical
          \draw [stamp majorline ] (            0  ,0) -- (            0  ,\stampheight);
          \draw [stamp majorline ] (  \stampwidth  ,0) -- (  \stampwidth  ,\stampheight);
        }
    }
}

% Compute the number of stamps that can fit in printable area
\pgfmathsetmacro{\nx}{int((\textwidth +\stamphsep)/(\stampwidth +\stamphsep))}
\pgfmathsetmacro{\ny}{int((\textheight+\stampvsep)/(\stampheight+\stampvsep))}

% Set PDF properties
\extractcolorspec{stampcolora}{\stampcolora}
\extractcolorspec{stampcolorb}{\stampcolorb}
\def\pdftitle{\ny*\nx{} \gridname{} of \stampheight-by-\stampwidth{} squares}
\hypersetup{
  pdftitle=\pdftitle,
  pdfkeywords={%
      \ny*\nx{} \gridname,
      h,s: \stampheight{} | \stampvsep,
      w,s: \stampwidth{} | \stamphsep,
      c: \{\stampcolora; \stampcolorb\}
    },
  pdfsubject={https://github.com/maverickwoo/paperpad-templates},
  pdfauthor={Maverick Woo}
}

\input{body}
